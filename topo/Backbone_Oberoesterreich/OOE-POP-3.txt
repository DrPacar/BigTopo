# BGP Attribute fuer Pfadsteuerung zu ROK via STN (AP 3.2)
# LOCAL_PREF 200 fuer ROK-Prefix (10.0.0.0/16) von STN-Neighbor
# -> OOE bevorzugt den Pfad ueber Backbone Steiermark zu HQ Rossauer Kaserne
ip prefix-list PREF_ROK seq 5 permit 10.0.0.0/16

route-map RM_FROM_STN permit 10
 match ip address prefix-list PREF_ROK
 set local-preference 200

route-map RM_FROM_STN permit 20

# BGP Konfiguration (AP 3.2)
# Confederation Main-AS 65006 | Sub-AS 65062 (IS-IS Area 49.0002)
# iBGP Client von OOE-BB-3 (Route Reflector) via Loopback
# eBGP zu STN-POP-5 (AS 65007) via 172.16.8.1
router bgp 65062
 bgp confederation identifier 65006
 bgp confederation peers 65061
 bgp router-id 10.6.0.13
 bgp log-neighbor-changes
 !
 neighbor 10.6.0.3 remote-as 65062
 neighbor 10.6.0.3 update-source Loopback0
 neighbor 10.6.0.3 description OOE-BB-3_iBGP-RR
 !
 neighbor 172.16.8.1 remote-as 65007
 neighbor 172.16.8.1 description STN-POP-5_eBGP
 neighbor 172.16.8.1 route-map RM_FROM_STN in
 !
 address-family ipv4
  network 10.6.0.0 mask 255.255.0.0
  neighbor 10.6.0.3 activate
  neighbor 172.16.8.1 activate
 exit-address-family
